#!/bin/bash
# Script: Kubernetes Worker Node Setup
# Run this on each Worker Node

set -e

echo "=== Disabling swap ==="
sudo swapoff -a
sudo sed -i '/ swap / s/^/#/' /etc/fstab

echo "=== Loading required kernel modules ==="
cat <<EOF | sudo tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF

sudo modprobe overlay
sudo modprobe br_netfilter

echo "=== Configuring sysctl parameters ==="
cat <<EOF | sudo tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                = 1
EOF

sudo sysctl -w net.ipv4.ip_forward=1
sudo sysctl --system

echo "=== Installing required dependencies ==="
sudo apt update -y
sudo apt install -y \
  curl \
  gnupg2 \
  software-properties-common \
  apt-transport-https \
  ca-certificates

echo "=== Adding containerd (Docker) repository ==="
sudo curl -fsSL https://download.docker.com/linux/ubuntu/gpg | \
  sudo gpg --dearmour -o /etc/apt/trusted.gpg.d/docker.gpg

sudo chmod 755 /etc/apt/trusted.gpg.d/docker.gpg

sudo add-apt-repository \
  "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"

echo "=== Installing containerd ==="
sudo apt update -y
sudo apt install -y containerd.io

echo "=== Configuring containerd to use systemd cgroup driver ==="
sudo containerd config default | sudo tee /etc/containerd/config.toml >/dev/null 2>&1
sudo sed -i 's/SystemdCgroup = false/SystemdCgroup = true/g' /etc/containerd/config.toml

echo "=== Enabling and starting containerd ==="
sudo systemctl restart containerd
sudo systemctl enable containerd

echo "=== Adding Kubernetes repository ==="
sudo mkdir -p /etc/apt/keyrings

curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.30/deb/Release.key | \
  sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg

sudo chmod 755 /etc/apt/keyrings/kubernetes-apt-keyring.gpg

echo "deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] \
https://pkgs.k8s.io/core:/stable:/v1.30/deb/ /" | \
  sudo tee /etc/apt/sources.list.d/kubernetes.list

echo "=== Installing kubeadm, kubelet, and kubectl ==="
sudo apt update -y
sudo apt install -y kubelet kubeadm kubectl

echo "=== Holding Kubernetes package versions ==="
sudo apt-mark hold kubelet kubeadm kubectl

echo "=== Final package verification ==="
sudo apt update -y
sudo apt list --upgradable

echo "=== Kubernetes Worker Node prerequisites completed successfully ==="
